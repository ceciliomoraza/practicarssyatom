public class PuntoDistancia{
  // Coordenadas del Punto en 2D.
  public double x;
  public double y;
  public double distancia( PuntoDistancia Param ){
    double distancia_calculada = 
      Math.sqrt(  Math.pow((Param.x-x) ,2) + 
                  Math.pow((Param.y-y) ,2) );
    return distancia_calculada;
  }
}
